The this keyword is confusing

  • Based on the usage, this refers to different things
    As a function
    		  var whatsThis = function() {
			console.log(this, this === window);
		  };

          whatsThis(); // window, true
    As a method
              var myObject = {
            whatsThis: function() {
              console.log(this, this === myObject);
            }
          };

           myObject.whatsThis(); // myObject, true
    As a Constructor
              var MyConstructor = function() {
            console.log(this);
          };

          var myObject = new MyConstructor(); // MyConstructor
    As a function on a function
              var myObject = {};
          myObject.whatsThis = function () {
            function whatsThat() {
              console.log(this);
            };

            whatsThat();
          };

          myObject.whatsThis(); // window
