The this keyword is still confusing

Let's look at that last example
      var myObject = {};
      myObject.whatsThis = function () {
        function whatsThat() {
          console.log(this);
        };

        whatsThat();
      };

      myObject.whatsThis(); // window
This is easy to get around
      var myObject = {};
      myObject.whatsThis = function () {
        var that = this;
        function whatsThat() {
          console.log(this);
          console.log(that);
        };

        whatsThat();
      };

      myObject.whatsThis(); // window
                            // myObject
